7、ES6中函数参数的默认值
    概述：
    ES允许给函数的参数赋初始值；
        
    代码示例及相关说明：
    //1. 形参初始值 具有默认值的参数, 一般位置要靠后(潜规则)
    function add(a, b, c = 10) {
      return a + b + c;
    }
    let result = add(1, 2);
    console.log(result); // 13

    //2. 与解构赋值结合
    // 注意这里参数是一个对象
    function connect({ host = "127.0.0.1", username, password, port }) {
      console.log(host);
      console.log(username);
      console.log(password);
      console.log(port);
    }
    connect({
      host: "atguigu.com",
      username: "root",
      password: "root",
      port: 3306,
    });
